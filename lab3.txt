CREATE TABLE Faculties (
    FacID INT PRIMARY KEY IDENTITY(1,1),
    FacName NVARCHAR(50) NOT NULL
);

CREATE TABLE StudyForms (
    FormID INT PRIMARY KEY IDENTITY(1,1),
    FormType NVARCHAR(50) NOT NULL
);

CREATE TABLE StudyPlans (
    PlanID INT PRIMARY KEY IDENTITY(1,1),
    CourseNum INT NOT NULL,
    FacID INT NOT NULL,
    FormID INT NOT NULL,
    TotalHours INT,
    PractHours INT,
    CONSTRAINT FK_Plans_Faculties FOREIGN KEY (FacID) REFERENCES Faculties(FacID),
    CONSTRAINT FK_Plans_Forms FOREIGN KEY (FormID) REFERENCES StudyForms(FormID)
);

CREATE TABLE Applicants (
    PersonID INT PRIMARY KEY IDENTITY(1,1),
    LastName NVARCHAR(25) NOT NULL,
    FirstName NVARCHAR(25) NOT NULL,
    Patronymic NVARCHAR(25),
    BirthDate DATE,
    EnterDate DATE,
    AvgScore FLOAT
);

CREATE TABLE Admission (
    PersonID INT NOT NULL,
    PlanID INT NOT NULL,
    CONSTRAINT PK_Admission PRIMARY KEY (PersonID, PlanID),
    CONSTRAINT FK_Admission_Applicants FOREIGN KEY (PersonID) REFERENCES Applicants(PersonID),
    CONSTRAINT FK_Admission_Plans FOREIGN KEY (PlanID) REFERENCES StudyPlans(PlanID)
);

SELECT a.LastName, a.FirstName, a.Patronymic, f.FacName, p.CourseNum 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
WHERE sf.FormType = 'Заочная' AND DATEDIFF(YEAR, a.BirthDate, GETDATE()) < 37;

SELECT f.FacName, COUNT(ad.PersonID) AS StudCount 
FROM Faculties f
JOIN StudyPlans p ON f.FacID = p.FacID
JOIN Admission ad ON p.PlanID = ad.PlanID
GROUP BY f.FacName;

SELECT sf.FormType, COUNT(ad.PersonID) AS StudCount 
FROM StudyForms sf
JOIN StudyPlans p ON sf.FormID = p.FormID
JOIN Admission ad ON p.PlanID = ad.PlanID
GROUP BY sf.FormType;

SELECT f.FacName, AVG(DATEDIFF(YEAR, a.BirthDate, GETDATE())) AS AvgAge 
FROM Faculties f
JOIN StudyPlans p ON f.FacID = p.FacID
JOIN Admission ad ON p.PlanID = ad.PlanID
JOIN Applicants a ON ad.PersonID = a.PersonID
GROUP BY f.FacName;

SELECT a.LastName, a.FirstName, a.EnterDate, f.FacName, p.CourseNum, sf.FormType 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
WHERE a.Patronymic IS NULL;

SELECT TOP 1 f.FacName, COUNT(a.PersonID) AS StudCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
WHERE YEAR(a.EnterDate) = 2015
GROUP BY f.FacName
ORDER BY StudCount DESC;

SELECT f.FacName, sf.FormType, COUNT(a.PersonID) AS StudCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
WHERE YEAR(a.EnterDate) = 2014
GROUP BY f.FacName, sf.FormType;

SELECT DISTINCT f.FacName 
FROM Faculties f
JOIN StudyPlans p ON f.FacID = p.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
WHERE sf.FormType = 'Заочная';

SELECT f.FacName, sf.FormType, p.CourseNum 
FROM Faculties f
JOIN StudyPlans p ON f.FacID = p.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
ORDER BY f.FacName, sf.FormType, p.CourseNum;

SELECT f.FacName, sf.FormType, COUNT(ad.PersonID) AS StudCount 
FROM Faculties f
JOIN StudyPlans p ON f.FacID = p.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
JOIN Admission ad ON p.PlanID = ad.PlanID
GROUP BY f.FacName, sf.FormType
ORDER BY f.FacName;

SELECT f.FacName, sf.FormType, COUNT(a.PersonID) AS StudCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
WHERE p.CourseNum IN (1, 3)
GROUP BY f.FacName, sf.FormType;

SELECT f.FacName, p.CourseNum, COUNT(a.PersonID) AS ForeignCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
WHERE a.Patronymic IS NULL
GROUP BY f.FacName, p.CourseNum
ORDER BY f.FacName, p.CourseNum;

SELECT f.FacName, p.CourseNum, COUNT(a.PersonID) AS StudCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
WHERE a.AvgScore > 7.5
GROUP BY f.FacName, p.CourseNum
ORDER BY f.FacName, p.CourseNum;

SELECT f.FacName, sf.FormType, COUNT(a.PersonID) AS StudCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
WHERE DATEDIFF(YEAR, a.BirthDate, GETDATE()) > 45
GROUP BY f.FacName, sf.FormType
ORDER BY f.FacName;

SELECT f.FacName, sf.FormType, p.CourseNum, COUNT(a.PersonID) AS StudCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
JOIN StudyForms sf ON p.FormID = sf.FormID
WHERE DATEDIFF(YEAR, a.BirthDate, GETDATE()) < 27
GROUP BY f.FacName, sf.FormType, p.CourseNum
ORDER BY f.FacName, sf.FormType, p.CourseNum;

SELECT f.FacName, COUNT(a.PersonID) AS StudCount 
FROM Applicants a
JOIN Admission ad ON a.PersonID = ad.PersonID
JOIN StudyPlans p ON ad.PlanID = p.PlanID
JOIN Faculties f ON p.FacID = f.FacID
WHERE a.LastName LIKE 'С%'
GROUP BY f.FacName
ORDER BY f.FacName;
