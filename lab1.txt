CREATE TABLE GROUPS (
    id_group VARCHAR(10) PRIMARY KEY,
    name_group VARCHAR(100) NOT NULL,
    course INT NOT NULL
);

CREATE TABLE SUBJECTS (
    id_subject INT IDENTITY(1,1) PRIMARY KEY,
    name_subject VARCHAR(150) NOT NULL,
    hours INT NOT NULL
);

CREATE TABLE STUDENTS (
    id_student INT IDENTITY(1,1) PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    middle_name VARCHAR(50),
    birth_date DATE NOT NULL,
    id_group VARCHAR(10)
);

CREATE TABLE PLAN (
    id_group VARCHAR(10),
    id_subject INT,
    PRIMARY KEY (id_group, id_subject)
);
